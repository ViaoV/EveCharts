\newcommand {\row} [2] {\SetRowColor{#1}#2\tn}

\newcommand {\headercell} [2] {
  \multicolumn{#1}{l}{\bfseries \small \MakeUppercase{#2}}
}
\newcommand {\cardheader} [2] {
  \row{CardHeader}{
    \multicolumn{#1}{c}{\bf\textcolor{white}{#2}}
  }
  \SetRowColor{white}
}

\newcommand {\card} [2] {
  \begin{tabularx}{\columnwidth}{X}
  \SetRowColor{CardHeader}
  \bf\textcolor{white}{#1} \tn
  \SetRowColor{white}
  #2
  \end{tabularx}
  \par\addvspace{1.3em}
}

\newcommand {\cardthree} [2] {
  \begin{tabularx}{\columnwidth}{X X X}
    \cardheader{3}{#1}
    #2
  \end{tabularx}
  \par\addvspace{1.3em}
}
